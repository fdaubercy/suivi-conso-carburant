- [ ] **Step 3 : Écrire le test a11y (rapport non bloquant)**

Create `tests/e2e/a11y.spec.js` :
```js
import { test } from '@playwright/test';
import AxeBuilder from '@axe-core/playwright';

const VUES = [
  { nom: 'saisie',     hash: '#/' },
  { nom: 'stats',      hash: '#/stats' },
  { nom: 'historique', hash: '#/historique' },
];

for (const vue of VUES) {
  test(`a11y ${vue.nom}`, async ({ page }, testInfo) => {
    await page.goto('/' + vue.hash);
    await page.waitForLoadState('networkidle');
    const results = await new AxeBuilder({ page })
      .withTags(['wcag2a', 'wcag2aa'])
      .analyze();
    const graves = results.violations.filter(v => ['critical', 'serious'].includes(v.impact));
    for (const v of graves) {
      testInfo.annotations.push({ type: 'a11y', description: `${v.id} (${v.impact}) ×${v.nodes.length} — ${v.help}` });
      console.warn(`[a11y:${vue.nom}] ${v.id} (${v.impact}) ×${v.nodes.length} — ${v.helpUrl}`);
    }
    console.warn(`[a11y:${vue.nom}] ${graves.length} violation(s) grave(s), ${results.violations.length} au total.`);
    // NON bloquant : on ne fait jamais échouer le test (W79 = visibilité, pas gate).
  });
}
```

- [ ] **Step 6 : Brancher `DebouncedRebuild`**

Modifier `DebouncedRebuild` (`modFiltres`) :
```vba
Dim newSig As String: newSig = ComputeRebuildSignature()
Select Case ClassifyFilterDelta(ReadLastSignature(), newSig)
    Case 0 ' rsNone
        SetStatus "Dashboard : a jour (aucun changement)."
        Exit Sub
    Case 1 ' rsTargeted
        Application.Run "RebuildBlocks", g_TargetedBlocks
    Case Else ' rsFull
        Application.Run "CreerGraphiquesWeb", True
End Select
WriteLastSignature newSig
```
Conserver le debounce et le mode silencieux.
